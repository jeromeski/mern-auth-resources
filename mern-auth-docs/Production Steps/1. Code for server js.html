Code for server.js (React)

Before we deploy our react app to production, there are some optimization that can be done for better performance and ease of running it live in server. You should not use npm start with create-react-app in production  because that command is made to be used for development mode.


Copy this block of code. You will need it in next lecture.


const express = require('express');
const compression = require('compression');
const path = require('path');
const app = express();

app.use(compression());
app.use(express.static(path.join(__dirname, 'build')));

app.get('*', function(req, res) {
    res.sendFile(path.join(__dirname, 'build', 'index.html'));
});

const PORT = process.env.PORT || 3000;

app.listen(PORT, () => {
    console.log(`App is running on port ${PORT}`);
});
